77  void Scene_Object::update_transform()
78  {
79  rotation += constantRotation; //Each time is entered in this method is added to the object rotation the constant rotation vector
80  Scale_Matrix3f scaleMatrix(scale);
81  Translation_Matrix3f positionMatrix(position);
82  Rotation_Matrix3f rotationX, rotationY, rotationZ;
83 
84  rotationX.set< Rotation_Matrix3f::AROUND_THE_X_AXIS >(rotation[X]);
85  rotationY.set< Rotation_Matrix3f::AROUND_THE_Y_AXIS >(rotation[Y]);
86  rotationZ.set< Rotation_Matrix3f::AROUND_THE_Z_AXIS >(rotation[Z]);
87 
88  //Creation of the unified transformation matrix :
89 
90  globalTransform = localTransform = positionMatrix * rotationX * rotationY * rotationZ * scaleMatrix;
91 
92  if (transformParent)
93  {
94  Transform_Matrix3f parent = *transformParent;
95  globalTransform = parent * globalTransform;
96  }
97 
98  update_global_attributes(); // Can`t extract the position properly for the moment
