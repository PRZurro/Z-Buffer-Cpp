121  void Model::update_vertex_buffers(Camera * activeCamera, Light * inputLight, float ambientalLightI)
122  {
123  tvPositions.resize(ovPositions.size());
124 
125  Matrix41f lightVector;
126 
127  if (!inputLight)
128  lightVector = Matrix41f(Vector4f({1.f, -1.f, 0.f, 0.f}));
129  else
130  lightVector = Matrix41f(inputLight->get_direction());
131 
132  Matrix44f cameraCoordinates = activeCamera->look_at(); // get the camera coordinates
133 
134  Vector4f lightV = cameraCoordinates * lightVector; // cache: calculate the light vector from the camera coordinates
135  vec4 lightglmV = normalize(vec4(lightV[X], lightV[Y], lightV[Z], lightV[W]));
136 
137  Matrix44f modelToCameraTransformation = cameraCoordinates * Matrix44f(globalTransform); // cache: get the model camera coordinates
138  Matrix44f modelProjectedTransformation = Matrix44f(activeCamera->get_projection_matrix()) * modelToCameraTransformation; // cache: pre-calculate the projected transformation
139 
140  for (size_t i = 0, numberOfIndices = ovPositions.size(); i < numberOfIndices; ++i)
141  {
142  //---------------------------------------Light calculation----------------------------------------------------
143  Vector4f normal = modelToCameraTransformation * Matrix41f(ovNormals[i]); //Calculate the transformed normal vector
144 
145  vec4 glmNormal({normal[X], normal[Y], normal[Z], normal[W]});
146  float intensity = std::max(dot(lightglmV, glmNormal), 0.f) + ambientalLightI;
147  vertexIntensities[i] = std::min(intensity, 1.f);
148 
149  tvNormals[i] = normal;
150 
151  //---------------------------------------Projected vertex coordinates calculation-----------------------------
152  tvPositions[i] = modelProjectedTransformation * Matrix41f(ovPositions[i]); //Calculate the transformed vertex position
153  }
154  }
